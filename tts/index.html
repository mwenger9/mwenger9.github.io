def speak(text):
    """Speak the text with gTTs in French

        Examples:
            >>>speak("Bonjour")
            say "Bonjour" in French

        Args:
            text (String): text to speak to the user in French
    """
    # Creating and starting a thread to execute the _speak function
    thread = threading.Thread(target=_speak(text, 'fr'))
    thread.start()
    thread.join()
