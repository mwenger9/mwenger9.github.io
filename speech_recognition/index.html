def findword(w):

    Examples:
        >>>findword("Hello World !")("Hello")
        something
        >>>findword("Hello World !")("Earth")
        None
        >>>findword("Hello World !")("Earth") is None
        True

    Args:
        w (String): word to find

    Return:
        None if the word in not in the specific String"""

    return re.compile(r'\b({0})\b'.format(w), flags=re.IGNORECASE).search

@contextlib.contextmanager
def ignore_stderr():
    """Ignore annoying errors"""
    devnull = os.open(os.devnull, os.O_WRONLY)
    old_stderr = os.dup(2)
    sys.stderr.flush()
    os.dup2(devnull, 2)
    os.close(devnull)
    try:
        yield
    finally:
        os.dup2(old_stderr, 2)
        os.close(old_stderr)

def listen(waiting):
    """Listen the user vocal entry in French with speech recognition of Google Speech Recognition.

        Examples:
            >>>listen(True)
            "Appuyez sur entrée pour parler"
            and listen
            >>>listen(False)
            just listen directy

        Args:
            waiting (boolean): if True wait for the user to press Enter

        Returns:
            sentence (String): what the user say
        """

    if debug:
        # only debug when cant use microphone and speaker
        # print("DEBUG MODE")
        return input("Veuillez saisir : ")
    else:
        request = "start"
        with ignore_stderr():
            while request != "stop":
                # wait for input
                inp = "ajwvzhx"
                if waiting:
                    while inp != "":
                        inp = input("Appuyez sur entrée pour parler\n")

                # audio reco
                r = sr.Recognizer()
                with sr.Microphone() as source:
                    print("En attente d'une instruction...")
                    audio = r.listen(source)

                try:
                    request = r.recognize_google(audio, language='fr-FR')
                    print(request)
                    return request
                except sr.UnknownValueError:
                    print("Je n'ai pas bien compris")
                    speak("Je n'ai pas bien compris")
                except sr.RequestError as e:
                    print("Could not request results from Google Speech Recognition service; {0}".format(e))
                    request = "stop"

def listen_whisper():
    """Listen with Whisper
    Recognition takes place when the user presses the space bar, and stops when he presses it again

    Examples:
        >>>listen_whisper()
        Wait for the user to press the space bar

    Return:
        sentence (String): the last word of what is recognized
        """

    result = "test"

    frames = []
    recording = False
    first_recognition_done = False

    def on_press(key):
        nonlocal recording, frames, first_recognition_done
        if key == pynput.keyboard.Key.space:
            if not recording:
                recording = True
                print("Recording started.")
                frames = []  # Initialize frames list for new recording
                try:
                    stream = p.open(format=FORMAT,
                                    channels=CHANNELS,
                                    rate=RATE,
                                    input=True,
                                    frames_per_buffer=CHUNK,
                                    stream_callback=callback)  # Pass callback function
                except Exception as e:
                    print("Error opening audio stream: {}".format(e))
                    recording = False  # Set recording back to False on error
            else:
                recording = False
                print("Recording stopped.")

                # Save the recorded audio
                wf = wave.open(WAVE_OUTPUT_FILENAME, 'wb')
                wf.setnchannels(CHANNELS)
                wf.setsampwidth(p.get_sample_size(FORMAT))
                wf.setframerate(RATE)
                wf.writeframes(b''.join(frames))
                wf.close()

                # Process the recorded audio and return the recognized text
                if not first_recognition_done:
                    first_recognition_done = True
                    # Stop the listener after the first recognition
                    listener.stop()
                    return

    def callback(in_data, frame_count, time_info, status):
        if recording:
            frames.append(in_data)
        return in_data, pyaudio.paContinue

    # Initialize PyAudio
    p = pyaudio.PyAudio()

    # Create keyboard listener
    listener = pynput.keyboard.Listener(on_press=on_press)
    listener.start()

    print("Press Space to start/stop recording.")

    # Wait indefinitely for key events (no timeout)
    listener.join()

    # Close PyAudio
    p.terminate()

    print("Recording finished.")
    model = loading.model
    # Transcribe the recorded audio
    result = model.transcribe(WAVE_OUTPUT_FILENAME)
    print("hadi", result["text"])

    return result["text"]
