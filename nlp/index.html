def generate_word_cloud(corpus,keep_stopwords : bool = False):
    """
    Generate a word cloud visualization from a corpus of documents.

    Args:
    - corpus: A list of strings, where each string represents a document in the corpus.

    Returns:
    None (plots the word cloud visualization).
    """
    # Concatenate all documents into a single string
    if not keep_stopwords:
        corpus = list(map(remove_stopwords,corpus))
    text = ' '.join(corpus)

    # Generate word cloud
    wordcloud = WordCloud(width=800, height=400, background_color='white').generate(text)

    # Plot word cloud
    plt.figure(figsize=(10, 6))
    plt.imshow(wordcloud, interpolation='bilinear')
    plt.axis('off')  # Hide axis
    plt.title('Word Cloud Visualization')
    plt.show()

def get_sentiment_score(text):
    """
    Calculate sentiment score for a word or a sentence using TextBlob.
    Returns a value between -1 (negative sentiment) and 1 (positive sentiment).
    """
    blob = TextBlob(text, pos_tagger=PatternTagger(), analyzer=PatternAnalyzer())
    print(blob.sentiment[0])
    return blob.sentiment[0]

def get_top_n_ngram(corpus,gram=2, n=20,keep_stopwords=False):

    """
    This function takes a corpus of text (a list of string, each string representing one document) 
    and will return the top n (20 by default) n-gram (Bigram by default) appearing in the corpus

    Args:
        corpus (list(str)): the list of document (each document is stored as one string)
        gram (int): the number of gram used : 2 is for bigram, 3 is for trigram and so on..
        n (int) : The max number of ngram kept for the ranking. By default 20 the top 20 ngram occurring will be returned.
        keep_stopwords (bool): Keep the stopwords or not, false by default so the stopwords will be removed

    Returns:
        df (pd.Series): A serie containing each n-gram and its occurrence

    """

    if not keep_stopwords:
        corpus = list(map(remove_stopwords,corpus))

    vec = CountVectorizer(ngram_range=(gram, gram),stop_words="english").fit(corpus)
    bag_of_words = vec.transform(corpus)
    sum_words = bag_of_words.sum(axis=0) 
    words_freq = [(word, sum_words[0, idx]) for word, idx in vec.vocabulary_.items()]
    words_freq =sorted(words_freq, key = lambda x: x[1], reverse=True)
    common_words = words_freq[:n]


    df = pd.DataFrame(common_words, columns = ['Text' , 'count'])

    return df.groupby('Text').sum()['count'].sort_values(ascending=False)

def get_top_n_words(corpus, n=20,keep_stopwords=False):

    """
    This function takes a corpus of text (a list of string, each string representing one document) 
    and will return the top n (20 by default) occuring words in the corpus

    Args:
	corpus (list(str)): the list of document (each document is stored as one string)
    keep_stopwords (bool): Keep the stopwords or not, false by default so the stopwords will be removed


     Returns:
        df (pd.Series): A serie containing the top 20 most occurring words. It will be returned as a Serie containing each word and their number of occurrences.
    """

    if not keep_stopwords:
        corpus = list(map(remove_stopwords,corpus))

    vec = CountVectorizer(stop_words = 'english').fit(corpus)
    bag_of_words = vec.transform(corpus)
    sum_words = bag_of_words.sum(axis=0) 
    words_freq = [(word, sum_words[0, idx]) for word, idx in vec.vocabulary_.items()]
    words_freq =sorted(words_freq, key = lambda x: x[1], reverse=True)
    common_words = words_freq[:n]
    df = pd.DataFrame(common_words, columns = ['Text' , 'count'])
    return df.groupby('Text').sum()['count'].sort_values(ascending=False)

def plot_words_with_score_heat(zipped_word_score_list):
    """
    Function used to plot words with scores describing them.
    The scores will be used to create a colored circle around the words in the plot. The color of the circles vary from red (score of -1.0) to green (score of 1.0)

    Input example: [('Je',0.0),('suis',0.0),('content',0.7)] -> will plot 'content' in a green circle because it is close to 1.0

    Args:
        zipped_word_score_list (tuple(word,float)): a list of tuple containing each word and their score (a value between -1.0 and 1.0) in form of tuples.

    Returns:
        None

    """

    num_words = len(zipped_word_score_list)

    plt.figure(figsize=(15, 3))  # Larger figure size
    for i, (word, polarity) in enumerate(zipped_word_score_list):
        # Calculate color gradient from red to green based on polarity
        intensity = abs(polarity)
        r = intensity if polarity < 0 else 0
        g = intensity if polarity > 0 else 0
        b = 0

        # Plot word
        plt.text(i, 0.5, word, ha='center', va='center', fontsize=12) 

        # Plot colored circle on top of the word
        circle = plt.Circle((i, 0.5), radius=0.3, color=(r, g, b), alpha=0.5)
        plt.gca().add_patch(circle)

    # Set plot limits and remove axes

    plt.xlim(-0.5, num_words - 0.5)
    plt.ylim(0, 1)
    plt.axis('off')
    plt.title('Polarity Plot')
    plt.show()

def visualize_embeddings(embeddings):
    """
    Visualize embeddings in a 2D plot with object names displayed at the projected locations.

    Args:
    - embeddings: A list of tuples where each tuple contains a string (name of the object) and a vector (embedding).

    Returns:
    None (plots the visualization).
    """
    # Extract embeddings and object names
    objects, vectors = zip(*embeddings)

    # Convert vectors to numpy array
    vectors = np.array(vectors)

    # Perform PCA to project embeddings into 2D space
    pca = PCA(n_components=2)
    projected_vectors = pca.fit_transform(vectors)

    # Plot 2D visualization
    plt.figure(figsize=(10, 8))
    plt.scatter(projected_vectors[:, 0], projected_vectors[:, 1], marker='o', color='b')

    # Display object names at the projected locations
    for i, (x, y) in enumerate(projected_vectors):
        plt.text(x, y, objects[i], fontsize=10, ha='right', va='bottom')

    # Set plot title and labels
    plt.title('Embedding Visualization')
    plt.xlabel('Principal Component 1')
    plt.ylabel('Principal Component 2')
    plt.grid(True)
    plt.show()
